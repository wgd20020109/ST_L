# 第3章：运算符与表达式

## 3.1 表达式基础

### 什么是表达式？

表达式是由操作数（变量、常量、函数调用）和运算符组成的组合，计算后产生一个值。

```st
(* 简单表达式 *)
a := 10;              (* 字面量表达式 *)
b := x + 5;           (* 算术表达式 *)
c := a > b;           (* 关系表达式 *)
result := (a + b) * c; (* 复合表达式 *)

(* 表达式求值顺序 *)
result := 2 + 3 * 4;  (* 结果是 14，不是 20 *)
result := (2 + 3) * 4; (* 结果是 20 *)
```

## 3.2 算术运算符

### 基本算术运算符

| 运算符 | 说明 | 示例 | 结果 |
|--------|------|------|------|
| + | 加法 | 5 + 3 | 8 |
| - | 减法 | 5 - 3 | 2 |
| * | 乘法 | 5 * 3 | 15 |
| / | 除法 | 15 / 4 | 3（整数除法） |
| MOD | 取模（余数） | 15 MOD 4 | 3 |
| ** | 幂运算 | 2 ** 3 | 8 |

```st
VAR
    a, b, c : INT;
    result : REAL;
END_VAR

(* 基本算术运算 *)
a := 10;
b := 3;
c := a + b;          (* c = 13 *)
c := a - b;          (* c = 7 *)
c := a * b;          (* c = 30 *)
c := a / b;          (* c = 3，整数除法 *)
c := a MOD b;        (* c = 1，余数 *)

(* 浮点数除法 *)
result := INT_TO_REAL(a) / INT_TO_REAL(b);  (* result = 3.333... *)

(* 幂运算 *)
c := 2 ** 10;        (* c = 1024 *)
```

### 算术运算注意事项

```st
VAR
    x, y, z : INT;
END_VAR

(* 整数除法 vs 浮点数除法 *)
x := 10;
y := 4;
z := x / y;          (* z = 2，不是 2.5 *)
z := x MOD y;        (* z = 2，余数 *)

(* 避免除零错误 *)
IF y <> 0 THEN
    result := x / y;
ELSE
    result := 0;     (* 处理除零情况 *)
END_IF

(* 运算顺序（优先级） *)
x := 2 + 3 * 4;      (* 先乘后加: x = 14 *)
x := (2 + 3) * 4;   (* 括号优先: x = 20 *)
```

## 3.3 关系运算符

### 比较运算符

| 运算符 | 说明 | 示例 |
|--------|------|------|
| = | 等于 | a = b |
| <> | 不等于 | a <> b |
| < | 小于 | a < b |
| > | 大于 | a > b |
| <= | 小于等于 | a <= b |
| >= | 大于等于 | a >= b |

```st
VAR
    temperature : INT := 85;
    minTemp : INT := 0;
    maxTemp : INT := 100;
    status : BOOL;
END_VAR

(* 关系运算示例 *)
status := temperature > minTemp;     (* TRUE *)
status := temperature < maxTemp;     (* TRUE *)
status := temperature = 85;          (* TRUE *)
status := temperature <> 100;        (* TRUE *)
status := temperature >= 80;         (* TRUE *)
status := temperature <= 90;          (* TRUE *)

(* 复合条件 *)
status := (temperature >= minTemp) AND (temperature <= maxTemp);

(* 浮点数比较（注意精度问题） *)
VAR
    x : REAL := 0.1;
    y : REAL := 0.1;
    equal : BOOL;
    epsilon : REAL := 1e-6;
END_VAR

(* 直接比较浮点数可能有精度问题 *)
equal := ABS(x - y) < epsilon;  (* 使用epsilon容差比较 *)
```

## 3.4 逻辑运算符

### 布尔逻辑运算符

| 运算符 | 说明 | 示例 |
|--------|------|------|
| AND | 逻辑与 | a AND b |
| OR | 逻辑或 | a OR b |
| XOR | 逻辑异或 | a XOR b |
| NOT | 逻辑非 | NOT a |

```st
VAR
    motorOn : BOOL := TRUE;
    brakeReleased : BOOL := TRUE;
    emergencyStop : BOOL := FALSE;
    canStart : BOOL;
    signalA, signalB, result : BOOL;
END_VAR

(* 逻辑与 (AND) - 所有条件都为TRUE时结果为TRUE *)
canStart := motorOn AND brakeReleased;  (* TRUE AND TRUE = TRUE *)
canStart := motorOn AND emergencyStop;  (* TRUE AND FALSE = FALSE *)

(* 逻辑或 (OR) - 任一条件为TRUE时结果为TRUE *)
canStart := canStart OR emergencyStop;  (* FALSE OR FALSE = FALSE *)

(* 逻辑异或 (XOR) - 条件不同时结果为TRUE *)
signalA := TRUE;
signalB := FALSE;
result := signalA XOR signalB;   (* TRUE XOR FALSE = TRUE *)
result := signalA XOR signalA;   (* TRUE XOR TRUE = FALSE *)

(* 逻辑非 (NOT) - 取反 *)
canStart := NOT emergencyStop;  (* NOT FALSE = TRUE *)

(* 复杂逻辑表达式 *)
canStart := (motorOn AND brakeReleased) AND NOT emergencyStop;
```

### 逻辑运算符的优先级

```
优先级从高到低: NOT > AND > XOR > OR

示例解析:
NOT a OR b AND c XOR d OR e
=
((NOT a) OR ((b AND c) XOR d)) OR e
```

## 3.5 位运算符

### 位操作运算符

| 运算符 | 说明 | 示例 |
|--------|------|------|
| & 或 AND | 位与 | 2#1010 & 2#1100 = 2#1000 |
| \| 或 OR | 位或 | 2#1010 \| 2#1100 = 2#1110 |
| XOR | 位异或 | 2#1010 XOR 2#1100 = 2#0110 |
| NOT | 位取反 | NOT 2#1010 = 2#0101 |
| SHL | 左移 | 2#1010 SHL 2 = 2#101000 |
| SHR | 右移 | 2#1010 SHR 1 = 2#101 |

```st
VAR
    data1 : BYTE := 2#10101010;
    data2 : BYTE := 2#11001100;
    result : BYTE;
    shiftResult : BYTE;
END_VAR

(* 位与 (AND) - 两位都为1时结果为1 *)
result := data1 & data2;
(* 10101010 & 11001100 = 10001000 *)

(* 位或 (OR) - 任一位为1时结果为1 *)
result := data1 OR data2;
(* 10101010 OR 11001100 = 11101110 *)

(* 位异或 (XOR) - 位不同时结果为1 *)
result := data1 XOR data2;
(* 10101010 XOR 11001100 = 01100110 *)

(* 位取反 (NOT) *)
result := NOT data1;
(* NOT 10101010 = 01010101 *)

(* 左移 (SHL) *)
shiftResult := data1 SHL 2;
(* 10101010 SHL 2 = 10101000 *)
(* 等同于乘以4 *)

(* 右移 (SHR) *)
shiftResult := data1 SHR 2;
(* 10101010 SHR 2 = 00101010 *)
(* 等同于除以4 *)
```

### 位运算实战应用

```st
VAR
    controlWord : WORD;
    enableBit : BOOL;
    modeBits : BYTE;
END_VAR

(* 提取特定位 *)
controlWord := 16#ABCD;
enableBit := (controlWord & 16#0001) <> 0;  (* 检查最低位 *)
modeBits := BYTE((controlWord SHR 8) AND 16#0F);  (* 高字节低4位 *)

(* 设置特定位 *)
controlWord := controlWord OR 16#0001;   (* 置位最低位 *)
controlWord := controlWord AND NOT 16#0001; (* 复位最低位 *)

(* 切换特定位 *)
controlWord := controlWord XOR 16#0001;  (* 翻转最低位 *)
```

## 3.6 字符串运算符

```st
VAR
    str1 : STRING(20) := 'Hello';
    str2 : STRING(20) := 'World';
    result : STRING(40);
END_VAR

(* 字符串连接 *)
result := CONCAT(str1, str2);           (* 'HelloWorld' *)
result := CONCAT(str1, ' ');           (* 'Hello ' *)
result := CONCAT(result, str2);        (* 'Hello World' *)

(* 部分PLC支持的操作 *)
result := LEFT(str1, 3);               (* 'Hel' *)
result := RIGHT(str1, 2);              (* 'lo' *)
result := MID(str1, 3, 2);             (* 'ell' *)
```

## 3.7 赋值语句

### 简单赋值

```st
VAR
    a, b : INT;
    x : REAL;
END_VAR

(* 简单赋值 *)
a := 10;
b := a;
x := 3.14;

(* 链式赋值（CODESYS支持） *)
// a := b := c := 10;  (* 部分编译器支持 *)
```

### 复合赋值

```st
VAR
    counter : INT := 0;
    accumulator : INT := 100;
END_VAR

(* 增量/减量 *)
counter := counter + 1;   (* 自增1 *)
counter := counter - 1;   (* 自减1 *)

(* 复合赋值运算符（部分编译器支持） *)
// counter += 1;   (* 等同于 counter := counter + 1 *)
// counter -= 5;   (* 等同于 counter := counter - 5 *)
// accumulator *= 2; (* 等同于 accumulator := accumulator * 2 *)
```

### 条件赋值

```st
VAR
    speed : INT := 0;
    direction : INT := 1;  (* 1=正转, -1=反转 *)
    outputSpeed : INT;
END_VAR

(* 条件赋值方法1：使用IF *)
IF speed > 0 THEN
    outputSpeed := speed * direction;
ELSE
    outputSpeed := 0;
END_IF

(* 条件赋值方法2：使用选择表达式（部分编译器支持） *)
// outputSpeed := SEL(speed > 0, 0, speed * direction);
// outputSpeed := speed > 0 ? speed * direction : 0;
```

## 3.8 表达式综合示例

### PID控制器核心计算

```st
(* 完整的PID计算表达式 *)
FUNCTION_BLOCK SimplePID
VAR_INPUT
    setpoint : REAL;
    processValue : REAL;
    kp : REAL := 1.0;
    ki : REAL := 0.1;
    kd : REAL := 0.05;
    enable : BOOL := TRUE;
END_VAR
VAR_OUTPUT
    output : REAL;
END_VAR
VAR
    integral : REAL := 0;
    prevError : REAL := 0;
    sampleTime : REAL := 0.1;  (* 100ms采样 *)
END_VAR

BEGIN
    IF enable THEN
        (* 计算当前误差 *)
        error := setpoint - processValue;

        (* 比例项 *)
        proportional := kp * error;

        (* 积分项 *)
        integral := integral + ki * error * sampleTime;

        (* 微分项 *)
        derivative := kd * (error - prevError) / sampleTime;

        (* PID输出 *)
        output := proportional + integral + derivative;

        (* 保存当前误差 *)
        prevError := error;
    ELSE
        (* 禁用时清零输出 *)
        output := 0;
        integral := 0;
        prevError := 0;
    END_IF;
END_FUNCTION_BLOCK
```

### 状态机条件判断

```st
VAR
    state : INT := 0;
    sensor1, sensor2, sensor3 : BOOL;
    timerDone : BOOL;
    command : INT;
END_VAR

(* 复杂条件判断示例 *)
IF state = 0 AND sensor1 THEN
    state := 1;
ELSIF state = 1 AND sensor2 THEN
    state := 2;
ELSIF state = 2 AND timerDone THEN
    state := 3;
ELSIF state = 3 AND (sensor1 OR sensor2 OR sensor3) THEN
    state := 0;
END_IF

(* 位掩码判断 *)
command := 16#0F;
IF (state AND command) = command THEN
    (* 处理匹配的情况 *)
END_IF
```

## 3.9 运算符优先级速查表

| 优先级 | 运算符 | 说明 |
|--------|--------|------|
| 1 | () | 括号 |
| 2 | ** | 幂运算 |
| 3 | NOT, - (取负) | 单目运算符 |
| 4 | *, /, MOD | 乘除取模 |
| 5 | +, - | 加减 |
| 6 | <, >, <=, >= | 比较 |
| 7 | =, <> | 等于/不等于 |
| 8 | AND, & | 逻辑与/位与 |
| 9 | XOR | 逻辑异或/位异或 |
| 10 | OR | 逻辑或/位或 |

**建议**：当不确定优先级时，使用括号明确指定运算顺序。

### 实践任务

### 常见错误

| 错误 | 原因 | 解决方法 |
|------|------|----------|
| 除法结果为0 | 整数除法截断 | 使用REAL类型或检查除数 |
| 浮点数比较失败 | 精度问题 | 使用epsilon容差比较 |
| 逻辑运算符混淆 | AND/OR优先级 | 使用括号明确顺序 |
| 位运算与逻辑运算混淆 | 符号相同但含义不同 | 注意区分&和AND |
