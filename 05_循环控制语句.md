# 第5章：循环控制语句

## 5.1 循环控制概述

循环语句允许程序重复执行一段代码，是处理批量数据、累加计算和状态机遍历的重要工具。

```
ST语言支持的循环类型
├── FOR循环    - 已知迭代次数
├── WHILE循环  - 条件为真时继续
├── REPEAT循环 - 直到条件满足
└── EXIT       - 退出循环
```

## 5.2 FOR循环

### 基本语法

```st
FOR <变量> := <起始值> TO <结束值> BY <步长> DO
    <语句>;
END_FOR

(* BY <步长> 是可选的，默认步长为1 *)
```

### 简单FOR循环

```st
VAR
    i : INT;
    sum : INT := 0;
END_VAR

(* 计算1+2+3+...+100 *)
FOR i := 1 TO 100 DO
    sum := sum + i;
END_FOR
(* sum = 5050 *)
```

### 指定步长的FOR循环

```st
VAR
    i : INT;
    evenSum : INT := 0;
    reverseSum : INT := 0;
END_VAR

(* 步长为2：计算偶数和 *)
FOR i := 2 TO 100 BY 2 DO
    evenSum := evenSum + i;
END_FOR
(* evenSum = 2550 *)

(* 逆序遍历 *)
FOR i := 100 TO 1 BY -1 DO
    reverseSum := reverseSum + i;
END_FOR
(* reverseSum = 5050 *)
```

### FOR循环遍历数组

```st
VAR
    i : INT;
    measurements : ARRAY[1..5] OF INT := [10, 20, 30, 40, 50];
    total : INT := 0;
    average : REAL;
END_VAR

(* 计算数组元素总和 *)
FOR i := 1 TO 5 DO
    total := total + measurements[i];
END_FOR

(* 计算平均值 *)
average := INT_TO_REAL(total) / 5.0;
(* average = 30.0 *)
```

### 嵌套FOR循环

```st
VAR
    i, j : INT;
    matrix : ARRAY[1..3, 1..3] OF INT;
    sum : INT := 0;
END_VAR

(* 初始化3x3矩阵 *)
FOR i := 1 TO 3 DO
    FOR j := 1 TO 3 DO
        matrix[i, j] := i * 10 + j;
    END_FOR
END_FOR

(* 遍历矩阵求和 *)
FOR i := 1 TO 3 DO
    FOR j := 1 TO 3 DO
        sum := sum + matrix[i, j];
    END_FOR
END_FOR
(* sum = 90 *)
```

## 5.3 WHILE循环

### 基本语法

```st
WHILE <条件> DO
    <语句>;
END_WHILE

(* 当条件为TRUE时重复执行 *)
```

### WHILE循环示例

```st
VAR
    counter : INT := 0;
    factorial : DINT := 1;
    n : INT := 5;
END_VAR

(* 计算5的阶乘 *)
WHILE counter < n DO
    counter := counter + 1;
    factorial := factorial * counter;
END_WHILE
(* factorial = 120 *)
```

### WHILE vs FOR

```st
VAR
    i : INT;
    sum : INT := 0;
END_VAR

(* FOR: 已知迭代次数 *)
FOR i := 1 TO 10 DO
    sum := sum + i;
END_FOR

(* WHILE: 迭代次数不确定 *)
i := 1;
WHILE sum < 100 DO
    sum := sum + i;
    i := i + 1;
END_WHILE
(* 循环直到sum >= 100 *)
```

### WHILE循环处理链表（模拟）

```st
TYPE
    Node : STRUCT
        value : INT;
        next : INT;  (* 下一个节点的索引，0表示无 *)
    END_STRUCT
END_TYPE

VAR
    nodes : ARRAY[1..10] OF Node;
    currentIndex : INT := 1;
    totalSum : INT := 0;
END_VAR

(* 初始化链表 *)
nodes[1].value := 10; nodes[1].next := 2;
nodes[2].value := 20; nodes[2].next := 3;
nodes[3].value := 30; nodes[3].next := 0;  (* 链表结尾 *)

(* 遍历链表 *)
WHILE currentIndex <> 0 DO
    totalSum := totalSum + nodes[currentIndex].value;
    currentIndex := nodes[currentIndex].next;
END_WHILE
(* totalSum = 60 *)
```

## 5.4 REPEAT循环

### 基本语法

```st
REPEAT
    <语句>;
UNTIL <条件>
END_REPEAT

(* 先执行一次，然后检查条件 *)
(* 条件为FALSE时继续循环 *)
```

### REPEAT循环示例

```st
VAR
    i : INT := 1;
    sum : INT := 0;
END_VAR

(* REPEAT至少执行一次 *)
REPEAT
    sum := sum + i;
    i := i + 1;
UNTIL sum > 100
END_REPEAT
(* i = 15, sum = 105 *)
```

### WHILE vs REPEAT对比

```st
VAR
    x : INT := 100;
END_VAR

(* WHILE: 条件初始为FALSE，不执行 *)
x := 100;
WHILE x < 10 DO
    x := x + 1;
END_WHILE
(* x = 100 (未进入循环) *)

(* REPEAT: 至少执行一次 *)
x := 100;
REPEAT
    x := x + 1;
UNTIL x >= 10
END_REPEAT
(* x = 101 (执行了一次) *)
```

## 5.5 EXIT语句

### 提前退出循环

```st
VAR
    i : INT;
    target : INT := 37;
    found : BOOL := FALSE;
    index : INT := 0;
    data : ARRAY[1..10] OF INT := [5, 12, 23, 37, 45, 56, 67, 78, 89, 95];
END_VAR

(* 查找元素，找到后提前退出 *)
FOR i := 1 TO 10 DO
    IF data[i] = target THEN
        found := TRUE;
        index := i;
        EXIT;  (* 找到目标，提前退出循环 *)
    END_IF
END_FOR

(* found = TRUE, index = 4 *)
```

### 多层循环中使用EXIT

```st
VAR
    i, j : INT;
    matrix : ARRAY[1..5, 1..5] OF INT;
    found : BOOL := FALSE;
    row, col : INT;
END_VAR

(* 初始化矩阵，查找第一个0 *)
FOR i := 1 TO 5 DO
    FOR j := 1 TO 5 DO
        IF matrix[i, j] = 0 THEN
            found := TRUE;
            row := i;
            col := j;
            EXIT;  (* 只退出内层循环 *)
        END_IF
    END_FOR

    (* 如果找到，可以在外层循环检查found状态 *)
    IF found THEN
        EXIT;  (* 找到后也退出外层循环 *)
    END_IF
END_FOR
```

## 5.6 CONTINUE语句（继续下一次迭代）

```st
VAR
    i : INT;
    sum : INT := 0;
    count : INT := 0;
END_VAR

(* 计算数组中正数的和，跳过负数和零 *)
FOR i := 1 TO 10 DO
    IF data[i] <= 0 THEN
        CONTINUE;  (* 跳过本次循环剩余代码 *)
    END_IF

    sum := sum + data[i];
    count := count + 1;
END_FOR

(* 不使用CONTINUE的等效写法 *)
FOR i := 1 TO 10 DO
    IF data[i] > 0 THEN
        sum := sum + data[i];
        count := count + 1;
    END_IF
END_FOR
```

## 5.7 RETURN语句

```st
FUNCTION FindMax : INT
VAR_INPUT
    data : ARRAY[1..10] OF INT;
    length : INT;
END_VAR
VAR
    i : INT;
    maxVal : INT;
END_VAR

(* 边界检查 *)
IF length <= 0 THEN
    FindMax := 0;
    RETURN;  (* 提前返回 *)
END_IF

(* 查找最大值 *)
maxVal := data[1];
FOR i := 2 TO length DO
    IF data[i] > maxVal THEN
        maxVal := data[i];
    END_IF
END_FOR

FindMax := maxVal;
END_FUNCTION
```

## 5.8 循环模式实战案例

### 案例1：冒泡排序

```st
FUNCTION_BLOCK BubbleSort
VAR_INPUT
    enable : BOOL;
END_VAR
VAR_IN_OUT
    data : ARRAY[1..20] OF INT;
    length : INT;
END_VAR
VAR
    i, j : INT;
    temp : INT;
    sorted : BOOL;
END_VAR

BEGIN
    IF NOT enable THEN
        RETURN;
    END_IF

    (* 冒泡排序 *)
    FOR i := 1 TO length - 1 DO
        sorted := TRUE;
        FOR j := 1 TO length - i DO
            IF data[j] > data[j + 1] THEN
                (* 交换 *)
                temp := data[j];
                data[j] := data[j + 1];
                data[j + 1] := temp;
                sorted := FALSE;
            END_IF
        END_FOR

        (* 如果一轮没有交换，已排序完成 *)
        IF sorted THEN
            EXIT;
        END_IF
    END_FOR
END_FUNCTION_BLOCK
```

### 案例2：查找最大/最小值

```st
FUNCTION_BLOCK MinMaxFinder
VAR_INPUT
    data : ARRAY[1..50] OF INT;
    count : INT;
END_VAR
VAR_OUTPUT
    minimum : INT;
    maximum : INT;
    average : REAL;
    minIndex : INT;
    maxIndex : INT;
END_VAR
VAR
    i : INT;
    sum : DINT;
END_VAR

BEGIN
    IF count <= 0 THEN
        minimum := 0;
        maximum := 0;
        average := 0.0;
        RETURN;
    END_IF

    (* 初始化 *)
    minimum := data[1];
    maximum := data[1];
    minIndex := 1;
    maxIndex := 1;
    sum := 0;

    (* 单次遍历找最大、最小、和 *)
    FOR i := 1 TO count DO
        sum := sum + data[i];

        IF data[i] < minimum THEN
            minimum := data[i];
            minIndex := i;
        END_IF

        IF data[i] > maximum THEN
            maximum := data[i];
            maxIndex := i;
        END_IF
    END_FOR

    average := DINT_TO_REAL(sum) / INT_TO_REAL(count);
END_FUNCTION_BLOCK
```

### 案例3：移动平均滤波器

```st
FUNCTION_BLOCK MovingAverageFilter
VAR_INPUT
    rawInput : INT;         (* 原始输入 *)
    enable : BOOL := TRUE;
    windowSize : INT := 5;   (* 窗口大小 *)
END_VAR
VAR_OUTPUT
    filteredOutput : INT;    (* 滤波输出 *)
END_VAR

VAR
    buffer : ARRAY[1..20] OF INT;
    i : INT;
    sum : DINT;
    oldestIndex : INT := 1;
END_VAR

BEGIN
    IF NOT enable THEN
        filteredOutput := rawInput;
        RETURN;
    END_IF

    (* 限制窗口大小 *)
    IF windowSize > 20 THEN
        windowSize := 20;
    END_IF

    (* 移除最旧的值，加入最新的值 *)
    sum := 0;
    FOR i := 1 TO windowSize DO
        sum := sum + buffer[i];
    END_FOR

    (* 更新缓冲区 *)
    buffer[oldestIndex] := rawInput;
    oldestIndex := oldestIndex + 1;
    IF oldestIndex > windowSize THEN
        oldestIndex := 1;
    END_IF

    (* 计算平均值 *)
    filteredOutput := DINT_TO_INT(sum / INT_TO_DINT(windowSize));
END_FUNCTION_BLOCK
```

### 案例4：查找算法

```st
(* 二分查找 *)
FUNCTION BinarySearch : INT
VAR_INPUT
    data : ARRAY[1..100] OF INT;
    count : INT;
    target : INT;
END_VAR
VAR
    low : INT := 1;
    high : INT;
    mid : INT;
    result : INT := 0;  (* 默认未找到 *)
END_VAR

BEGIN
    high := count;

    WHILE low <= high DO
        mid := (low + high) / 2;

        IF data[mid] = target THEN
            result := mid;    (* 找到 *)
            RETURN;
        ELSIF data[mid] < target THEN
            low := mid + 1;   (* 目标在右半部分 *)
        ELSE
            high := mid - 1;  (* 目标在左半部分 *)
        END_IF
    END_WHILE;

    BinarySearch := result;  (* 未找到返回0 *)
END_FUNCTION
```

## 5.9 循环性能注意事项

### 避免空循环

```st
VAR
    i : INT;
END_VAR

(* 不推荐：空循环 *)
FOR i := 1 TO 1000 DO
    ;  (* 什么都不做 *)
END_FOR

(* 推荐：设置起始值 *)
i := 1000;  (* 直接赋值 *)
```

### 循环体内的优化

```st
VAR
    i, sum1, sum2 : INT;
    factor : INT := 10;
    data : ARRAY[1..100] OF INT;
END_VAR

(* 不推荐：重复计算 *)
FOR i := 1 TO 100 DO
    sum1 := sum1 + data[i] * factor + 5;
    sum2 := sum2 + data[i] * factor - 5;
END_FOR

(* 推荐：提取公共表达式 *)
FOR i := 1 TO 100 DO
    sum1 := sum1 + data[i] * factor + 5;
    sum2 := sum2 + (data[i] * factor) - 5;  (* 只计算一次乘法 *)
END_FOR
```

### 早期退出

```st
VAR
    i : INT;
    hasNegative : BOOL := FALSE;
END_VAR

(* 查找负数，找到后提前退出 *)
FOR i := 1 TO 1000 DO
    IF data[i] < 0 THEN
        hasNegative := TRUE;
        EXIT;  (* 提前退出，避免不必要的迭代 *)
    END_IF
END_FOR
```

## 5.10 本章小结

### 循环类型对比

| 循环类型 | 适用场景 | 特点 |
|----------|----------|------|
| FOR | 已知迭代次数 | 最常用，可指定步长 |
| WHILE | 条件驱动 | 条件不满足不执行 |
| REPEAT | 至少执行一次 | 先执行后检查 |

### 控制语句

| 语句 | 作用 |
|------|------|
| EXIT | 退出当前循环 |
| CONTINUE | 跳到下一次迭代 |
| RETURN | 退出当前函数/功能块 |

### 最佳实践

1. **优先使用FOR**：已知迭代次数时首选
2. **边界检查**：使用WHILE前确保有退出条件
3. **避免死循环**：确保循环条件最终会变为FALSE
4. **提前退出**：找到目标后立即EXIT
5. **最小化循环体**：循环内只放必要代码
