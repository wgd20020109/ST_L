# 第2章：变量声明与数据类型

## 2.1 变量基础概念

### 什么是变量？

变量是用于存储数据的命名内存位置。在ST语言中，每个变量都有：
- **名称（Identifier）**：用于引用变量
- **数据类型（Data Type）**：决定变量可以存储什么类型的数据
- **值（Value）**：变量当前存储的数据
- **地址（Address）**：变量在内存中的位置（可寻址变量）

### 变量命名规则

```st
(* 合法的变量名 *)
MotorSpeed      (* 驼峰命名法 *)
motor_speed     (* 下划线命名法 *)
counter1        (* 带数字 *)
IO_Port_A       (* 含下划线和字母 *)

(* 非法的变量名（以数字开头） *)
// 1variable    (* 错误！不能以数字开头 *)
// my-var       (* 错误！不能含连字符 *)
// my.var       (* 错误！不能含点号 *)
```

### 变量声明的基本语法

```st
VAR
    variableName : dataType := initialValue;
END_VAR
```

## 2.2 基本数据类型

### 布尔型（BOOL）

存储 `TRUE` 或 `FALSE`，是PLC编程中最常用的类型。

```st
VAR
    motorRunning : BOOL;           (* 默认值 FALSE *)
    sensorActive : BOOL := TRUE;    (* 初始化为 TRUE *)
    alarmTriggered : BOOL := 0;     (* 也可以用0/1赋值 *)
END_VAR

(* 使用示例 *)
motorRunning := TRUE;          (* 启动电机 *)
sensorActive := (temperature > 100); (* 条件赋值 *)
alarmTriggered := motorRunning AND sensorActive;
```

### 整数类型

| 类型 | 位数 | 范围 | 说明 |
|------|------|------|------|
| SINT | 8位 | -128 ~ 127 | 短整数 |
| INT | 16位 | -32,768 ~ 32,767 | 整数 |
| DINT | 32位 | ±约21亿 | 双整数 |
| LINT | 64位 | 极大范围 | 长整数 |
| USINT | 8位 | 0 ~ 255 | 无符号短整数 |
| UINT | 16位 | 0 ~ 65,535 | 无符号整数 |
| UDINT | 32位 | 0 ~ 4,294,967,295 | 无符号双整数 |
| ULINT | 64位 | 极大范围 | 无符号长整数 |

```st
VAR
    (* 有符号整数 *)
    temperature : INT := 25;        (* 温度值 *)
    counter : DINT := 0;            (* 计数器 *)
    smallValue : SINT := -10;       (* 小范围值 *)

    (* 无符号整数 *)
    portAddress : UINT := 16#100;  (* 十六进制表示 *)
    byteData : USINT := 255;       (* 字节数据 *)
    wordData : UDINT := 16#FFFF;   (* 字数据 *)
END_VAR

(* 数值范围示例 *)
temperature := 32767;  (* INT最大值 *)
counter := -2147483648; (* DINT最小值 *)
```

### 浮点数类型

| 类型 | 位数 | 精度 | 范围 |
|------|------|------|------|
| REAL | 32位 | 约7位十进制 | ±1.175e-38 ~ ±3.402e+38 |
| LREAL | 64位 | 约15位十进制 | ±2.225e-308 ~ ±1.798e+308 |

```st
VAR
    pidKp : REAL := 1.5;           (* PID比例系数 *)
    pi : LREAL := 3.141592653589793; (* 圆周率高精度 *)
    voltage : REAL := 220.5;        (* 电压值 *)
    speed : LREAL := 1500.75;       (* 电机转速 *)
END_VAR

(* 指数表示法 *)
VAR
    small : REAL := 1.0e-6;        (* 0.000001 *)
    large : LREAL := 1.0e15;        (* 1000000000000000 *)
END_VAR
```

### 时间和日期类型

| 类型 | 说明 | 示例 |
|------|------|------|
| TIME | 时间持续时长 | T#5s, T#100ms, T#1h30m |
| DATE | 日期 | D#2024-01-15 |
| TIME_OF_DAY / TOD | 一天内的时间 | TOD#12:30:45 |
| DATE_AND_TIME / DT | 日期和时间 | DT#2024-01-15-12:30:45 |

```st
VAR
    delayTime : TIME := T#2s500ms;     (* 2.5秒延时 *)
    cycleTime : TIME := T#100ms;        (* 100毫秒周期 *)
    powerOnDate : DATE := D#2024-01-01; (* 电源开启日期 *)
    currentTime : TOD := TOD#08:30:00;   (* 当前时间 *)
    timestamp : DT := DT#2024-01-15-14:30:00; (* 时间戳 *)

    (* 时间常量的多种写法 *)
    time1 : TIME := T#10s;              (* 10秒 *)
    time2 : TIME := t#1m30s;            (* 1分30秒 *)
    time3 : TIME := T#1H_30M_20S;       (* 1小时30分20秒 *)
    time4 : TIME := TIME#10.5s;         (* 带小数 *)
END_VAR

(* 时间比较 *)
IF delayTime > T#5s THEN
    (* 延时大于5秒时执行 *)
END_IF
```

### 字符串类型（STRING）

```st
VAR
    deviceName : STRING(20) := 'Motor_A';      (* 最大20字符 *)
    message : STRING(50) := 'System Ready';     (* 最大50字符 *)
    emptyString : STRING(10);                   (* 空字符串 *)
    chineseText : WSTRING(20) := "中文文本";    (* 宽字符串 *)
END_VAR

(* 字符串操作（部分控制器支持） *)
deviceName := 'Pump_01';
message := 'Temperature: ';
```

### 字节和字类型

| 类型 | 说明 | 等同于 |
|------|------|--------|
| BYTE | 8位数据 | ARRAY[0..7] OF BOOL |
| WORD | 16位数据 | ARRAY[0..15] OF BOOL |
| DWORD | 32位数据 | ARRAY[0..31] OF BOOL |
| LWORD | 64位数据 | ARRAY[0..63] OF BOOL |

```st
VAR
    dataByte : BYTE := 2#10101010;     (* 二进制表示 *)
    dataWord : WORD := 16#FFA0;        (* 十六进制表示 *)
    dataDword : DWORD := 16#12345678;  (* 双字 *)
END_VAR

(* 位访问 *)
dataByte.0 := TRUE;   (* 设置最低位 *)
dataWord.7 := FALSE;  (* 清除某位 *)
```

## 2.3 变量声明的位置

### 程序级变量（VAR ... END_VAR）

```st
PROGRAM MotorControl
VAR
    (* 局部变量，仅在此程序内可见 *)
    motorSpeed : INT := 0;
    acceleration : INT := 10;
    targetSpeed : INT := 100;
    currentPosition : INT := 0;
END_VAR

BEGIN
    (* 主程序代码 *)
    IF motorSpeed < targetSpeed THEN
        motorSpeed := motorSpeed + acceleration;
    END_IF
END_PROGRAM
```

### 输入变量（VAR_INPUT）

```st
FUNCTION_BLOCK PIDController
VAR_INPUT
    (* 输入参数：由外部赋值 *)
    setpoint : REAL;           (* 设定值 *)
    processValue : REAL;        (* 过程值/反馈值 *)
    kp : REAL := 1.0;           (* 比例系数，可设默认值 *)
    ki : REAL := 0.1;           (* 积分系数 *)
    kd : REAL := 0.05;          (* 微分系数 *)
    enable : BOOL := TRUE;      (* 使能信号 *)
END_VAR

VAR_OUTPUT
    (* 输出参数：输出到外部 *)
    output : REAL;              (* 控制输出 *)
    error : REAL;               (* 当前误差 *)
END_VAR

VAR
    (* 内部状态：保持上次调用的值 *)
    integral : REAL := 0;
    lastError : REAL := 0;
END_VAR
END_FUNCTION_BLOCK
```

### 完整的功能块变量声明

```st
FUNCTION_BLOCK Counter
VAR_INPUT
    reset : BOOL;              (* 复位输入 *)
    increment : BOOL;          (* 增量触发 *)
    preset : INT := 100;       (* 预设值，默认100 *)
    direction : BOOL := TRUE; (* 计数方向，TRUE=递增 *)
END_VAR

VAR_OUTPUT
    currentValue : INT;        (* 当前计数值 *)
    done : BOOL;              (* 计数完成标志 *)
    overflow : BOOL;          (* 溢出标志 *)
END_VAR

VAR
    lastIncrement : BOOL;     (* 上次增量状态 *)
END_VAR
END_FUNCTION_BLOCK
```

### 全局变量（VAR_GLOBAL）

```st
(* 文件：GlobalVariables.gvl *)

VAR_GLOBAL
    (* 全局变量：所有POU都可以访问 *)
    systemRunning : BOOL := FALSE;
    emergencyStop : BOOL := FALSE;
    globalCounter : INT := 0;

    (* 常量 *)
    MAX_SPEED : INT := 3000;
    MIN_SPEED : INT := 0;
    PI : REAL := 3.14159;
END_VAR

(* 在程序中使用全局变量 *)
PROGRAM MainProgram
VAR
    localVar : INT := 100;
END_VAR

BEGIN
    (* 读取全局变量 *)
    IF systemRunning THEN
        localVar := globalCounter;
    END_IF

    (* 修改全局变量 *)
    globalCounter := localVar + 1;
END_PROGRAM
```

### 常量声明（VAR_CONSTANT）

```st
VAR
    (* 常量：在程序运行期间不能修改 *)
    CONST_1 : INT := 100;
    GRAVITY : REAL := 9.81;
    MAX_ITERATIONS : INT := 1000;
    DEVICE_ID : STRING(20) := 'PLC_MAIN_001';
END_VAR

VAR_CONSTANT
    (* 另一种常量声明方式 *)
    MAX_VOLTAGE : REAL := 24.0;
    MIN_VOLTAGE : REAL := 0.0;
    CALIBRATION_OFFSET : REAL := -0.5;
END_VAR
```

## 2.4 变量初始化

### 默认初始化值

| 数据类型 | 默认值 |
|----------|--------|
| BOOL | FALSE |
| 整数类型 | 0 |
| 浮点数类型 | 0.0 |
| 字符串 | '' (空字符串) |
| 指针 | NULL |
| 复杂类型 | 各元素按类型默认值 |

```st
VAR
    (* 显式初始化 *)
    value1 : INT := 10;            (* 初始化为10 *)
    flag : BOOL := TRUE;           (* 初始化为TRUE *)
    ratio : REAL := 1.5;            (* 初始化为1.5 *)

    (* 默认初始化（未指定值） *)
    value2 : INT;                  (* 默认为0 *)
    flag2 : BOOL;                  (* 默认为FALSE *)
END_VAR
```

### 数组和结构体初始化

```st
TYPE
    SensorConfig : STRUCT
        name : STRING(20);
        enabled : BOOL := TRUE;
        threshold : INT := 100;
    END_STRUCT
END_TYPE

VAR
    (* 数组初始化 *)
    measurements : ARRAY[1..5] OF INT := [1, 2, 3, 4, 5];
    flags : ARRAY[0..3] OF BOOL := [TRUE, FALSE, FALSE, TRUE];

    (* 结构体初始化 *)
    sensor1 : SensorConfig := (name := 'Temp_01', threshold := 150);

    (* 多维数组初始化 *)
    matrix : ARRAY[1..2, 1..3] OF INT := [1, 2, 3, 4, 5, 6];
END_VAR
```

## 2.5 变量的作用域

### 作用域规则

```
全局作用域 (Global)
├── VAR_GLOBAL 声明的变量
└── 所有程序/功能块可访问
                      │
        ┌─────────────┼─────────────┐
        ▼             ▼             ▼
┌───────────┐  ┌───────────┐  ┌───────────┐
│  程序A    │  │  程序B    │  │ 功能块X   │
│ 局部变量  │  │ 局部变量  │  │ 输入/输出 │
│ 局部变量  │  │ 局部变量  │  │ 内部状态  │
└───────────┘  └───────────┘  └───────────┘
```

### 作用域示例

```st
(* ========== GlobalVariables.gvl ========== *)
VAR_GLOBAL
    globalSpeed : INT := 0;    (* 全局变量 *)
END_VAR

(* ========== Program: MainControl ========== *)
PROGRAM MainControl
VAR
    mainCounter : INT := 0;   (* MainControl的局部变量 *)
END_VAR

VAR_INPUT
    inputValue : INT;         (* 输入参数 *)
END_VAR

BEGIN
    (* 可以访问：globalSpeed, mainCounter, inputValue *)
    globalSpeed := inputValue;
    mainCounter := mainCounter + 1;
END_PROGRAM

(* ========== Function: Calculate ========== *)
FUNCTION Calculate : INT
VAR_INPUT
    x : INT;
    y : INT;
END_VAR
VAR
    temp : INT;               (* 函数的局部变量 *)
END_VAR

BEGIN
    (* 可以访问：x, y, temp, globalSpeed *)
    temp := x * y;
    Calculate := temp + globalSpeed;
END_FUNCTION
```

## 2.6 可寻址变量（直接地址访问）

某些场景下需要直接访问物理地址（如访问特定内存区域）：

```st
VAR
    (* AT 关键字：将变量映射到物理地址 *)
    inputByte AT %IB0 : BYTE;     (* 输入字节0 *)
    outputByte AT %QB0 : BYTE;    (* 输出字节0 *)

    (* 位地址访问 *)
    motor1Run AT %QX1.0 : BOOL;  (* 输出字1的第0位 *)
    sensor1 AT %IX2.5 : BOOL;    (* 输入字2的第5位 *)

    (* 地址范围 *)
    dataArea AT %MD100 : DWORD;   (* 内存双字地址100 *)
END_VAR

(* 使用示例 *)
inputByte := 16#FF;          (* 写输出 *)
motor1Run := TRUE;            (* 启动电机1 *)
```

## 2.7 类型转换

### 隐式转换（自动转换）

较小的整数类型可以自动转换为较大的整数类型：

```st
VAR
    small : SINT := 10;
    medium : INT;
    large : DINT;
    floatValue : REAL;
END_VAR

(* 隐式转换（安全） *)
medium := small;     (* SINT → INT *)
large := medium;     (* INT → DINT *)
floatValue := medium; (* INT → REAL *)
```

### 显式转换（强制转换）

必须显式指定的类型转换：

```st
VAR
    realValue : REAL := 3.14159;
    intValue : INT;
    wordValue : WORD;
    byteValue : BYTE;
END_VAR

(* 显式类型转换 *)
intValue := INT(3.14159);      (* REAL → INT，结果为3 *)
intValue := REAL_TO_INT(realValue); (* 使用转换函数 *)
byteValue := LOW_BYTE(INT_TO_WORD(intValue)); (* 低字节提取 *)

(* 常用转换函数 *)
wordValue := INT_TO_WORD(intValue);
byteValue := WORD_TO_BYTE(wordValue); (* 取低位字节 *)
```

## 2.8 本章小结

### 核心要点

| 概念 | 说明 |
|------|------|
| 变量声明 | `VAR ... END_VAR` 块内声明 |
| 基本类型 | BOOL, INT/DINT, REAL/LREAL, TIME/DATE, STRING |
| 变量位置 | VAR（局部）、VAR_INPUT、VAR_OUTPUT、VAR_GLOBAL（全局） |
| 初始化 | 可以在声明时指定初始值 |
| 作用域 | 全局 → 程序/功能块 → 函数 |
| 类型转换 | 隐式（自动）或显式（强制） |

### 实践任务

1. 声明不同类型的变量并初始化
2. 创建包含输入、输出、内部变量的功能块
3. 使用全局变量在多个程序间共享数据
4. 练习类型之间的转换

### 常见错误

| 错误 | 原因 | 解决方法 |
|------|------|----------|
| 未声明变量 | 使用了未声明的变量名 | 检查变量声明 |
| 类型不匹配 | 给变量赋了错误类型的值 | 使用类型转换函数 |
| 超出范围 | 值超出数据类型的范围 | 选择更大的类型或检查逻辑 |
| 大小写错误 | 关键字或变量名大小写不一致 | ST不区分大小写，但建议统一风格 |
