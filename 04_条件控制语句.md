# 第4章：条件控制语句

## 4.1 条件控制概述

条件控制语句允许程序根据特定条件的满足与否来选择执行不同的代码路径。这是编程中最基本的控制结构之一。

```
条件为TRUE
    ↓
┌─────────────┐
│  IF 条件    │──NO──→ 跳过 THEN 块
│  THEN       │
│   代码...   │
└─────────────┘
```

## 4.2 IF语句

### 基本语法

```st
IF <条件> THEN
    <语句>;
END_IF
```

### 单分支IF

```st
VAR
    temperature : INT := 85;
    alarm : BOOL;
END_VAR

(* 简单IF语句 *)
IF temperature > 100 THEN
    alarm := TRUE;
END_IF
```

### IF-ELSE双分支

```st
VAR
    value : INT := 10;
    result : INT;
END_VAR

(* IF-ELSE 结构 *)
IF value > 0 THEN
    result := value;          (* 条件为TRUE时执行 *)
ELSE
    result := -value;         (* 条件为FALSE时执行 *)
END_IF
```

### IF-ELSIF-ELSE多分支

```st
VAR
    score : INT := 85;
    grade : STRING(2);
END_VAR

(* 多分支条件 *)
IF score >= 90 THEN
    grade := 'A';
ELSIF score >= 80 THEN
    grade := 'B';
ELSIF score >= 70 THEN
    grade := 'C';
ELSIF score >= 60 THEN
    grade := 'D';
ELSE
    grade := 'F';
END_IF
```

### 嵌套IF语句

```st
VAR
    age : INT := 25;
    income : REAL := 50000.0;
    category : INT;
END_VAR

(* 嵌套IF *)
IF age >= 18 THEN
    IF income > 100000.0 THEN
        category := 3;  (* 高收入成年人 *)
    ELSIF income > 50000.0 THEN
        category := 2;  (* 中等收入成年人 *)
    ELSE
        category := 1;  (* 低收入成年人 *)
    END_IF
ELSE
    category := 0;      (* 未成年 *)
END_IF
```

## 4.3 CASE语句

### 基本语法

```st
CASE <表达式> OF
    <值1>: <语句>;
    <值2>: <语句>;
    <值3>, <值4>: <语句>;  (* 多个值匹配 *)
    <值5>..<值10>: <语句>; (* 范围匹配 *)
ELSE
    <语句>;               (* 默认处理 *)
END_CASE
```

### 简单CASE示例

```st
VAR
    state : INT := 2;
    status : STRING(20);
END_VAR

CASE state OF
    0: status := 'Idle';
    1: status := 'Starting';
    2: status := 'Running';
    3: status := 'Stopping';
    4: status := 'Fault';
ELSE
    status := 'Unknown';
END_CASE
```

### 多值和范围匹配

```st
VAR
    month : INT := 3;
    days : INT;
END_VAR

CASE month OF
    1, 3, 5, 7, 8, 10, 12:  (* 大月 *)
        days := 31;
    4, 6, 9, 11:             (* 小月 *)
        days := 30;
    2:                       (* 二月 *)
        days := 28;
ELSE
    days := 0;
END_CASE

(* 范围匹配示例 *)
VAR
    score : INT := 75;
    level : INT;
END_VAR

CASE score OF
    90..100: level := 1;    (* 优秀 *)
    80..89:  level := 2;    (* 良好 *)
    70..79:  level := 3;    (* 中等 *)
    60..69:  level := 4;    (* 及格 *)
    0..59:   level := 5;    (* 不及格 *)
END_CASE
```

### CASE与枚举类型

```st
TYPE
    MotorState : (
        MOTOR_IDLE,
        MOTOR_STARTING,
        MOTOR_RUNNING,
        MOTOR_STOPPING,
        MOTOR_FAULT
    );
END_TYPE

VAR
    motorState : MotorState := MOTOR_IDLE;
    message : STRING(30);
END_VAR

(* CASE处理枚举类型 *)
CASE motorState OF
    MOTOR_IDLE:
        message := 'Motor is stopped';
    MOTOR_STARTING:
        message := 'Motor starting...';
    MOTOR_RUNNING:
        message := 'Motor running OK';
    MOTOR_STOPPING:
        message := 'Motor stopping...';
    MOTOR_FAULT:
        message := 'Motor fault detected!';
END_CASE
```

## 4.4 条件表达式（选择函数）

### SEL选择函数

```st
VAR
    flag : BOOL := TRUE;
    in1 : INT := 10;
    in2 : INT := 20;
    result : INT;
END_VAR

(* SEL: G=TRUE返回IN1，G=FALSE返回IN2 *)
result := SEL(flag, in2, in1);  (* flag=TRUE，返回10 *)

flag := FALSE;
result := SEL(flag, in2, in1);  (* flag=FALSE，返回20 *)
```

### MAX和MIN函数

```st
VAR
    a : INT := 15;
    b : INT := 8;
    c : INT := 20;
    maxVal : INT;
    minVal : INT;
END_VAR

(* MAX: 返回最大值 *)
maxVal := MAX(a, b);       (* 15 *)
maxVal := MAX(a, b, c);   (* 20 *)

(* MIN: 返回最小值 *)
minVal := MIN(a, b);       (* 8 *)
minVal := MIN(a, b, c);   (* 8 *)

(* 限幅应用 *)
VAR
    input : INT := 150;
    output : INT;
    upperLimit : INT := 100;
    lowerLimit : INT := 0;
END_VAR

output := MIN(input, upperLimit);  (* 限制上限 *)
output := MAX(output, lowerLimit); (* 限制下限 *)

(* 或使用单一表达式 *)
output := MIN(MAX(input, lowerLimit), upperLimit);
```

### LIMIT函数

```st
VAR
    input : INT := 150;
    limited : INT;
    lo : INT := 0;
    hi : INT := 100;
END_VAR

(* LIMIT: 将输入限制在[lo, hi]范围内 *)
limited := LIMIT(lo, input, hi);  (* 返回100 *)
```

## 4.5 实战案例

### 案例1：电机启停控制

```st
FUNCTION_BLOCK MotorControl
VAR_INPUT
    startButton : BOOL;    (* 启动按钮 *)
    stopButton : BOOL;     (* 停止按钮 *)
    faultInput : BOOL;     (* 故障信号 *)
END_VAR

VAR_OUTPUT
    motorOn : BOOL;        (* 电机运行输出 *)
    faultIndicator : BOOL; (* 故障指示 *)
END_VAR

VAR
    lastStart : BOOL := FALSE;
    lastStop : BOOL := FALSE;
END_VAR

BEGIN
    (* 边沿检测：仅在按钮从OFF变为ON时触发 *)
    IF startButton AND NOT lastStart THEN
        IF NOT faultInput THEN
            motorOn := TRUE;       (* 启动电机 *)
        END_IF
    END_IF;

    lastStart := startButton;

    (* 停止按钮检测 *)
    IF stopButton AND NOT lastStop THEN
        motorOn := FALSE;          (* 停止电机 *)
    END_IF;

    lastStop := stopButton;

    (* 故障处理 *)
    IF faultInput THEN
        motorOn := FALSE;          (* 故障时停止电机 *)
        faultIndicator := TRUE;    (* 置位故障指示 *)
    ELSE
        faultIndicator := FALSE;
    END_IF
END_FUNCTION_BLOCK
```

### 案例2：温度监控系统

```st
FUNCTION_BLOCK TemperatureMonitor
VAR_INPUT
    currentTemp : REAL;    (* 当前温度 *)
    highAlarmLimit : REAL := 80.0;
    lowAlarmLimit : REAL := 20.0;
    warningLimit : REAL := 70.0;
    resetButton : BOOL;
END_VAR

VAR_OUTPUT
    status : INT;          (* 0=正常, 1=警告, 2=高报警, 3=低报警 *)
    heaterOn : BOOL;
    coolerOn : BOOL;
    alarmBuzzer : BOOL;
END_VAR

VAR
    alarmActive : BOOL := FALSE;
END_VAR

BEGIN
    (* 温度状态判断 *)
    IF currentTemp >= highAlarmLimit THEN
        status := 2;               (* 高报警 *)
        alarmActive := TRUE;
    ELSIF currentTemp <= lowAlarmLimit THEN
        status := 3;               (* 低报警 *)
        alarmActive := TRUE;
    ELSIF currentTemp >= warningLimit THEN
        status := 1;               (* 警告 *)
    ELSE
        status := 0;               (* 正常 *)
    END_IF

    (* 报警输出 *)
    alarmBuzzer := alarmActive;

    (* 温度控制 *)
    IF currentTemp < (warningLimit - 5.0) THEN
        heaterOn := TRUE;
        coolerOn := FALSE;
    ELSIF currentTemp > (warningLimit + 5.0) THEN
        heaterOn := FALSE;
        coolerOn := TRUE;
    ELSE
        heaterOn := FALSE;
        coolerOn := FALSE;
    END_IF

    (* 报警复位 *)
    IF resetButton THEN
        alarmActive := FALSE;
    END_IF
END_FUNCTION_BLOCK
```

### 案例3：交通灯控制器

```st
TYPE
    LightState : (
        STATE_RED,
        STATE_RED_YELLOW,
        STATE_GREEN,
        STATE_YELLOW
    );
END_TYPE

FUNCTION_BLOCK TrafficLight
VAR_INPUT
    enable : BOOL;          (* 使能信号 *)
    autoMode : BOOL := TRUE; (* 自动/手动模式 *)
    manualButton : BOOL;   (* 手动切换按钮 *)
END_VAR

VAR_OUTPUT
    redLight : BOOL;
    yellowLight : BOOL;
    greenLight : BOOL;
    currentState : LightState;
END_VAR

VAR
    timer : TON;
    cycleStep : INT := 0;
END_VAR

BEGIN
    IF enable THEN
        (* 自动模式 *)
        IF autoMode THEN
            timer(IN := TRUE, PT := T#5s);

            IF timer.Q THEN
                timer(IN := FALSE);
                cycleStep := (cycleStep + 1) MOD 4;
            END_IF

            (* 状态转换 *)
            CASE cycleStep OF
                0:  (* 红灯 *)
                    currentState := STATE_RED;
                1:  (* 红灯+黄灯 *)
                    currentState := STATE_RED_YELLOW;
                2:  (* 绿灯 *)
                    currentState := STATE_GREEN;
                3:  (* 黄灯 *)
                    currentState := STATE_YELLOW;
            END_CASE
        ELSE
            (* 手动模式 *)
            IF manualButton THEN
                cycleStep := (cycleStep + 1) MOD 4;
            END_IF

            CASE cycleStep OF
                0:  currentState := STATE_RED;
                1:  currentState := STATE_RED_YELLOW;
                2:  currentState := STATE_GREEN;
                3:  currentState := STATE_YELLOW;
            END_CASE
        END_IF

        (* 灯光输出 *)
        redLight := (currentState = STATE_RED) OR (currentState = STATE_RED_YELLOW);
        yellowLight := (currentState = STATE_RED_YELLOW) OR (currentState = STATE_YELLOW);
        greenLight := (currentState = STATE_GREEN);
    ELSE
        (* 禁用状态：全部熄灭 *)
        redLight := FALSE;
        yellowLight := FALSE;
        greenLight := FALSE;
        currentState := STATE_RED;
    END_IF
END_FUNCTION_BLOCK
```

### 案例4：液位PID控制

```st
FUNCTION_BLOCK LevelPID
VAR_INPUT
    setpoint : REAL;        (* 液位设定值 *)
    actualLevel : REAL;     (* 实际液位 *)
    enable : BOOL;
END_VAR

VAR_OUTPUT
    valvePosition : REAL;  (* 阀门开度 0-100% *)
END_VAR

VAR
    kp : REAL := 2.0;
    ki : REAL := 0.5;
    integral : REAL := 0;
    lastError : REAL := 0;
    error : REAL;
    derivative : REAL;
    output : REAL;
    sampleTime : REAL := 0.1;
END_VAR

BEGIN
    IF enable THEN
        (* 计算误差 *)
        error := setpoint - actualLevel;

        (* 比例项 *)
        output := kp * error;

        (* 积分项 *)
        integral := integral + ki * error * sampleTime;

        (* 积分限幅：防止积分饱和 *)
        IF integral > 50.0 THEN integral := 50.0; END_IF
        IF integral < -50.0 THEN integral := -50.0; END_IF

        output := output + integral;

        (* 微分项 *)
        derivative := (error - lastError) / sampleTime;
        output := output + 0.1 * derivative;

        lastError := error;

        (* 输出限幅 *)
        IF output > 100.0 THEN
            output := 100.0;
        ELSIF output < 0.0 THEN
            output := 0.0;
        END_IF

        valvePosition := output;
    ELSE
        (* 禁用时关闭阀门 *)
        valvePosition := 0.0;
        integral := 0.0;
        lastError := 0.0;
    END_IF
END_FUNCTION_BLOCK
```

## 4.6 条件语句的技巧

### 布尔表达式的简化

```st
VAR
    a, b, c : BOOL;
END_VAR

(* 简化前 *)
IF a = TRUE AND b = TRUE THEN
    c := TRUE;
ELSE
    c := FALSE;
END_IF

(* 简化后：直接赋值 *)
c := a AND b;
```

### 多条件组合技巧

```st
VAR
    enable : BOOL;
    mode1, mode2, mode3 : BOOL;
    result : BOOL;
END_VAR

(* 所有条件必须满足 *)
result := enable AND mode1 AND mode2;

(* 任一条件满足即可 *)
result := enable OR mode1 OR mode2;

(* 复杂组合 *)
result := enable AND (mode1 OR mode2) AND NOT mode3;
```

### 短路求值

```st
VAR
    ptr : INT;  (* 指针或索引 *)
    data : ARRAY[1..10] OF INT;
    valid : BOOL;
END_VAR

(* 短路求值：前面的条件为FALSE时，后面的不执行 *)
IF (ptr > 0) AND (ptr <= 10) AND (data[ptr] > 0) THEN
    valid := TRUE;
END_IF

(* 在ST中，这是安全的写法 *)
```

## 4.7 本章小结

### 核心语法

| 语句 | 用途 |
|------|------|
| `IF ... THEN ... END_IF` | 单分支条件 |
| `IF ... THEN ... ELSE ... END_IF` | 双分支条件 |
| `IF ... ELSIF ... ELSIF ... ELSE ... END_IF` | 多分支条件 |
| `CASE ... OF ... END_CASE` | 多值选择 |
| `SEL(G, IN1, IN2)` | 二选一函数 |

### 选择IF还是CASE？

| 场景 | 推荐 |
|------|------|
| 条件是布尔表达式 | IF |
| 条件是范围判断 | IF |
| 条件是离散值选择（枚举、整数） | CASE |
| 多个值需要相同处理 | CASE（多值匹配） |
