# 第8章：数组与结构

## 8.1 数组基础

### 什么是数组？

数组是一组相同类型数据的集合，通过索引访问各个元素。

```
数组: temperatures
索引:    [1]    [2]    [3]    [4]    [5]
数据:   [23]   [25]   [22]   [24]   [26]
         ↑                            ↑
       temperatures[1]           temperatures[5]
```

### 数组声明语法

```st
VAR
    (* 一维数组 *)
    values : ARRAY[1..10] OF INT;
    flags : ARRAY[0..7] OF BOOL;

    (* 带初始化的数组 *)
    initValues : ARRAY[1..5] OF INT := [10, 20, 30, 40, 50];

    (* 多维数组 *)
    matrix : ARRAY[1..3, 1..3] OF REAL;

    (* 三维数组 *)
    cube : ARRAY[1..2, 1..2, 1..2] OF INT;
END_VAR
```

## 8.2 一维数组

### 基本操作

```st
VAR
    temperatures : ARRAY[1..7] OF INT;
    i : INT;
    sum : INT;
END_VAR

(* 赋值 *)
temperatures[1] := 23;
temperatures[2] := 25;
temperatures[7] := 28;

(* 读取 *)
sum := temperatures[1] + temperatures[2] + temperatures[7];

(* 遍历求和 *)
sum := 0;
FOR i := 1 TO 7 DO
    sum := sum + temperatures[i];
END_FOR
```

### 数组初始化

```st
VAR
    (* 部分初始化，未指定的元素为0 *)
    partial : ARRAY[1..5] OF INT := [1, 2];  (* [1,2,0,0,0] *)

    (* 全部初始化 *)
    full : ARRAY[1..4] OF BOOL := [TRUE, FALSE, TRUE, FALSE];

    (* 重复值初始化（部分编译器支持） *)
    zeros : ARRAY[1..10] OF INT := [10(0)];  (* 10个0 *)
    ones : ARRAY[1..5] OF INT := [5(1)];     (* 5个1 *)

    (* 字符串数组 *)
    dayNames : ARRAY[0..6] OF STRING(10) := 
        ['Sunday', 'Monday', 'Tuesday', 'Wednesday', 
         'Thursday', 'Friday', 'Saturday'];
END_VAR
```

### 数组边界检查

```st
VAR
    data : ARRAY[1..10] OF INT;
    index : INT;
END_VAR

(* 安全访问 - 边界检查 *)
index := getIndex();  (* 假设是动态获取的索引 *)

IF index >= 1 AND index <= 10 THEN
    data[index] := 100;
ELSE
    (* 处理越界情况 *)
    errorCode := 1;
END_IF

(* 或使用CLAMP函数限幅 *)
index := CLAMP(index, 1, 10);
data[index] := 100;
```

## 8.3 多维数组

### 二维数组声明与访问

```st
VAR
    (* 3行4列的矩阵 *)
    matrix : ARRAY[1..3, 1..4] OF INT;

    (* 棋盘示例 *)
    chessBoard : ARRAY[1..8, 1..8] OF INT;  (* 0=空, 1=白棋, 2=黑棋 *)

    i, j : INT;
END_VAR

(* 访问二维数组 *)
matrix[1, 1] := 1;
matrix[2, 3] := 5;
value := matrix[3, 4];

(* 遍历二维数组 *)
FOR i := 1 TO 3 DO
    FOR j := 1 TO 4 DO
        matrix[i, j] := i * 10 + j;
    END_FOR
END_FOR
```

### 矩阵初始化

```st
VAR
    (* 按行初始化二维数组 *)
    identity : ARRAY[1..3, 1..3] OF INT := 
        [1, 0, 0,
         0, 1, 0,
         0, 0, 1];

    (* 单位矩阵的另一写法 *)
    unitMatrix : ARRAY[1..3, 1..3] OF INT;

    (* 初始化为单位矩阵 *)
    i : INT;
END_VAR

FOR i := 1 TO 3 DO
    unitMatrix[i, i] := 1;
END_FOR
```

### 三维数组

```st
VAR
    (* 2层 x 3行 x 4列 *)
    volume : ARRAY[1..2, 1..3, 1..4] OF REAL;

    layer, row, col : INT;
END_VAR

(* 赋值 *)
volume[1, 1, 1] := 1.5;
volume[2, 3, 4] := 2.5;

(* 遍历 *)
FOR layer := 1 TO 2 DO
    FOR row := 1 TO 3 DO
        FOR col := 1 TO 4 DO
            volume[layer, row, col] := 0.0;
        END_FOR
    END_FOR
END_FOR
```

## 8.4 结构体基础

### 什么是结构体？

结构体是将不同类型数据组合成一个逻辑单元的用户自定义类型。

```
结构体: Point
├── x : INT    (坐标X)
├── y : INT    (坐标Y)
└── label : STRING(20)  (标签)

结构体: Motor
├── position : INT      (位置)
├── speed : INT         (速度)
├── enabled : BOOL      (使能)
└── fault : BOOL        (故障)
```

### 结构体声明

```st
TYPE
    (* 定义结构体类型 *)
    Point : STRUCT
        x : INT;
        y : INT;
    END_STRUCT;

    MotorStatus : STRUCT
        position : INT;
        speed : INT;
        enabled : BOOL;
        fault : BOOL;
    END_STRUCT;

    (* 嵌套结构体 *)
    RobotArm : STRUCT
        base : Point;           (* 使用已定义的结构体 *)
        joint1 : INT;
        joint2 : INT;
        joint3 : INT;
        toolAttached : BOOL;
    END_STRUCT;
END_TYPE

VAR
    p1 : Point;                  (* 声明结构体变量 *)
    motor1 : MotorStatus;
    arm1 : RobotArm;
END_VAR
```

### 结构体使用

```st
VAR
    start : Point;
    target : Point;
END_VAR

(* 赋值结构体成员 *)
start.x := 100;
start.y := 200;

target.x := start.x + 50;
target.y := start.y + 100;

(* 批量赋值 *)
start := (x := 0, y := 0);
target := (x := 500, y := 300);
```

## 8.6 实战案例

### 案例1：数据采集与统计

```st
TYPE
    Measurement : STRUCT
        timestamp : TIME;
        temperature : REAL;
        pressure : REAL;
        flowRate : REAL;
    END_STRUCT;

    MeasurementLog : STRUCT
        measurements : ARRAY[1..100] OF Measurement;
        count : INT := 0;
        maxCount : INT := 100;
    END_STRUCT;
END_TYPE

FUNCTION_BLOCK DataLogger
VAR_INPUT
    enable : BOOL;
    newTemp : REAL;
    newPressure : REAL;
    newFlow : REAL;
    reset : BOOL;
END_VAR
VAR_OUTPUT
    avgTemp : REAL;
    avgPressure : REAL;
    maxTemp : REAL;
    minTemp : REAL;
    sampleCount : INT;
END_VAR
VAR
    log : MeasurementLog;
    i : INT;
    tempSum : REAL;
END_VAR

BEGIN
    IF reset THEN
        log.count := 0;
        avgTemp := 0.0;
        avgPressure := 0.0;
        maxTemp := 0.0;
        minTemp := 0.0;
        RETURN;
    END_IF

    IF enable AND log.count < log.maxCount THEN
        (* 记录新数据 *)
        log.count := log.count + 1;
        log.measurements[log.count].timestamp := TIME();
        log.measurements[log.count].temperature := newTemp;
        log.measurements[log.count].pressure := newPressure;
        log.measurements[log.count].flowRate := newFlow;
    END_IF

    (* 统计计算 *)
    IF log.count > 0 THEN
        tempSum := 0.0;
        maxTemp := log.measurements[1].temperature;
        minTemp := log.measurements[1].temperature;

        FOR i := 1 TO log.count DO
            tempSum := tempSum + log.measurements[i].temperature;

            IF log.measurements[i].temperature > maxTemp THEN
                maxTemp := log.measurements[i].temperature;
            END_IF

            IF log.measurements[i].temperature < minTemp THEN
                minTemp := log.measurements[i].temperature;
            END_IF
        END_FOR

        avgTemp := tempSum / INT_TO_REAL(log.count);
    END_IF

    sampleCount := log.count;
END_FUNCTION_BLOCK
```

### 案例2：多轴运动控制

```st
TYPE
    AxisStatus : STRUCT
        position : DINT;        (* 当前位置 *)
        targetPosition : DINT; (* 目标位置 *)
        velocity : INT;        (* 当前速度 *)
        maxVelocity : INT;     (* 最大速度 *)
        acceleration : INT;    (* 加速度 *)
        enabled : BOOL;
        homed : BOOL;          (* 已回零 *)
        moving : BOOL;
        error : BOOL;
    END_STRUCT;

    MachineState : STRUCT
        axes : ARRAY[1..4] OF AxisStatus;
        emergencyStop : BOOL;
        cycleStartTime : TIME;
        cycleCount : INT;
    END_STRUCT;
END_TYPE

FUNCTION_BLOCK MotionController
VAR_INPUT
    machine : MachineState;
END_VAR
VAR_OUTPUT
    newMachine : MachineState;
END_VAR
VAR
    i : INT;
END_VAR

BEGIN
    newMachine := machine;

    FOR i := 1 TO 4 DO
        IF NOT newMachine.axes[i].enabled OR newMachine.emergencyStop THEN
            newMachine.axes[i].moving := FALSE;
            newMachine.axes[i].error := TRUE;
        ELSE
            IF newMachine.axes[i].homed THEN
                IF newMachine.axes[i].position < newMachine.axes[i].targetPosition THEN
                    newMachine.axes[i].moving := TRUE;
                    newMachine.axes[i].velocity := MIN(
                        newMachine.axes[i].velocity + newMachine.axes[i].acceleration,
                        newMachine.axes[i].maxVelocity
                    );
                ELSIF newMachine.axes[i].position > newMachine.axes[i].targetPosition THEN
                    newMachine.axes[i].moving := TRUE;
                    newMachine.axes[i].velocity := -MIN(
                        ABS(newMachine.axes[i].velocity) + newMachine.axes[i].acceleration,
                        newMachine.axes[i].maxVelocity
                    );
                ELSE
                    newMachine.axes[i].moving := FALSE;
                    newMachine.axes[i].velocity := 0;
                END_IF
            END_IF
        END_IF
    END_FOR
END_FUNCTION_BLOCK
```

### 案例3：配方管理系统

```st
TYPE
    Ingredient : STRUCT
        name : STRING(20);
        quantity : REAL;
        unit : STRING(10);
    END_STRUCT;

    RecipeStep : STRUCT
        stepNumber : INT;
        description : STRING(50);
        duration : TIME;
        temperature : REAL;
        speed : INT;
    END_STRUCT;

    Recipe : STRUCT
        id : INT;
        name : STRING(30);
        productCode : STRING(15);
        ingredients : ARRAY[1..10] OF Ingredient;
        steps : ARRAY[1..20] OF RecipeStep;
        ingredientCount : INT;
        stepCount : INT;
    END_STRUCT;
END_TYPE

VAR
    currentRecipe : Recipe;
    selectedRecipeIndex : INT;
    availableRecipes : ARRAY[1..5] OF Recipe;
END_VAR

(* 加载配方 *)
FUNCTION_BLOCK LoadRecipe
VAR_INPUT
    recipeId : INT;
END_VAR
VAR_OUTPUT
    recipe : Recipe;
    success : BOOL;
END_VAR
VAR
    i : INT;
END_VAR

BEGIN
    success := FALSE;

    (* 查找配方 *)
    FOR i := 1 TO 5 DO
        IF availableRecipes[i].id = recipeId THEN
            recipe := availableRecipes[i];
            success := TRUE;
            EXIT;
        END_IF
    END_FOR
END_FUNCTION_BLOCK
```

### 案例4：查找和排序

```st
(* 查找数组中的最大值及其索引 *)
FUNCTION FindMax : INT
VAR_INPUT
    data : ARRAY[1..100] OF INT;
    count : INT;
END_VAR
VAR_OUTPUT
    maxIndex : INT;
END_VAR
VAR
    i : INT;
    maxVal : INT;
END_VAR

BEGIN
    IF count <= 0 THEN
        FindMax := 0;
        maxIndex := 0;
        RETURN;
    END_IF;

    maxVal := data[1];
    maxIndex := 1;

    FOR i := 2 TO count DO
        IF data[i] > maxVal THEN
            maxVal := data[i];
            maxIndex := i;
        END_IF
    END_FOR;

    FindMax := maxVal;
END_FUNCTION

(* 冒泡排序 *)
FUNCTION_BLOCK BubbleSort
VAR_IN_OUT
    data : ARRAY[1..100] OF INT;
    count : INT;
END_VAR
VAR
    i, j, temp : INT;
    swapped : BOOL;
END_VAR

BEGIN
    FOR i := 1 TO count - 1 DO
        swapped := FALSE;
        FOR j := 1 TO count - i DO
            IF data[j] > data[j + 1] THEN
                temp := data[j];
                data[j] := data[j + 1];
                data[j + 1] := temp;
                swapped := TRUE;
            END_IF
        END_FOR;

        IF NOT swapped THEN
            EXIT;  (* 已排序完成 *)
        END_IF
    END_FOR
END_FUNCTION_BLOCK
```

### 案例5：FIFO队列

```st
TYPE
    QueueEntry : STRUCT
        timestamp : TIME;
        value : INT;
        priority : INT;
    END_STRUCT;
END_TYPE

FUNCTION_BLOCK FIFOQueue
VAR_INPUT
    push : BOOL;
    pop : BOOL;
    newValue : INT;
    priority : INT := 0;
END_VAR
VAR_OUTPUT
    frontValue : INT;
    queueFull : BOOL;
    queueEmpty : BOOL;
    itemCount : INT;
END_VAR
VAR
    queue : ARRAY[1..20] OF QueueEntry;
    head : INT := 1;
    tail : INT := 1;
    count : INT := 0;
    lastPush : BOOL := FALSE;
    lastPop : BOOL := FALSE;
END_VAR

BEGIN
    (* 入队 *)
    IF push AND NOT lastPush THEN
        IF count < 20 THEN
            queue[tail].timestamp := TIME();
            queue[tail].value := newValue;
            queue[tail].priority := priority;
            tail := (tail MOD 20) + 1;
            count := count + 1;
        END_IF
    END_IF;

    (* 出队 *)
    IF pop AND NOT lastPop AND count > 0 THEN
        frontValue := queue[head].value;
        head := (head MOD 20) + 1;
        count := count - 1;
    END_IF;

    (* 状态输出 *)
    queueFull := (count >= 20);
    queueEmpty := (count <= 0);
    itemCount := count;

    IF count > 0 THEN
        frontValue := queue[head].value;
    ELSE
        frontValue := 0;
    END_IF;

    lastPush := push;
    lastPop := pop;
END_FUNCTION_BLOCK
```

## 8.7 类型转换与数组

VAR

## 8.8 本章小结

### 数组要点

| 特性 | 说明 |
|------|------|
| 声明 | `ARRAY[下界..上界] OF 类型` |
| 多维 | `ARRAY[1..3, 1..3] OF REAL` |
| 初始化 | `[val1, val2, ...]` |
| 访问 | `array[index]` |

### 结构体要点

| 特性 | 说明 |
|------|------|
| 定义 | `TYPE ... STRUCT ... END_STRUCT END_TYPE` |
| 成员访问 | `struct.member` |
| 嵌套 | 结构体可包含其他结构体 |
| 数组 | 结构体数组 `ARRAY[1..n] OF StructType` |
